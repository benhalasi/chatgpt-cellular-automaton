import java.util.Arrays;

public class CellularAutomaton {
    private boolean[] currentState;
    private boolean[] nextState;
    private final int rule;

    public CellularAutomaton(int rule, int length) {
        this.rule = rule;
        currentState = new boolean[length];
        nextState = new boolean[length];

        // Set the initial state to all false
        Arrays.fill(currentState, false);
        currentState[length / 2] = true;  // Set the middle cell to true
    }

    public void step() {
        for (int i = 0; i < currentState.length; i++) {
            int left = (i == 0) ? currentState.length - 1 : i - 1;
            int right = (i == currentState.length - 1) ? 0 : i + 1;
            int index = (currentState[left] ? 4 : 0) + (currentState[i] ? 2 : 0) + (currentState[right] ? 1 : 0);
            nextState[i] = ((rule >> index) & 1) == 1;
        }

        boolean[] temp = currentState;
        currentState = nextState;
        nextState = temp;
    }

    public boolean[] getCurrentState() {
        return currentState;
    }
}
